#include <stdio.h>

void swapByValue(int a, int b) {
    int temp;
    temp = a;
    a = b;
    b = temp;
   
}


void swapByReference(int *x, int *y) {
    int temp;
    temp = *x;
    *x = *y;
    *y = temp;
}

int main() {
    int num1, num2;

   
    printf("Enter two numbers: ");
    scanf("%d %d", &num1, &num2);

 
    int a = num1, b = num2;
    int x = num1, y = num2;

   
    swapByValue(a, b);
   
    printf("After swapping without pointers: a = %d b = %d\n", b, a);

    
    swapByReference(&x, &y);
    printf("After swapping with pointers: x = %d y = %d\n", x, y);

    return 0;
}
