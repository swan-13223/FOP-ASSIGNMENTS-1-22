#include <stdio.h>
#include <string.h>

// Define the Structure EMPLOYEE [cite: 1, 5]
struct EMPLOYEE {
    char name[50];
    char designation[30];
    char gender[10];
    char doj[15]; // Date of Joining
    float salary;
};

// Function members to perform required operations [cite: 2]

void displayStats(struct EMPLOYEE emp[], int n) {
    int maleCount = 0, femaleCount = 0;

    // a) Total number of employees [cite: 2, 6]
    printf("\nTotal number of employees = %d", n);

    // b) Count of male and female employees [cite: 2, 6]
    for (int i = 0; i < n; i++) {
        if (strcasecmp(emp[i].gender, "Male") == 0) {
            maleCount++;
        } else if (strcasecmp(emp[i].gender, "Female") == 0) {
            femaleCount++;
        }
    }
    printf("\nMale employees = %d", maleCount);
    printf("\nFemale employees = %d", femaleCount);

    // c) Employees with salary more than 10,000 [cite: 2, 6]
    printf("\nEmployees with salary more than 10000: ");
    int foundSalary = 0;
    for (int i = 0; i < n; i++) {
        if (emp[i].salary > 10000) {
            printf("%s ", emp[i].name);
            foundSalary = 1;
        }
    }
    if (!foundSalary) printf("None");

    // d) Employees with designation "Asst Manager" [cite: 2, 6]
    printf("\nEmployees with designation AsstManager: ");
    int foundDesignation = 0;
    for (int i = 0; i < n; i++) {
        if (strcasecmp(emp[i].designation, "AsstManager") == 0 || 
            strcasecmp(emp[i].designation, "Asst Manager") == 0) {
            printf("%s ", emp[i].name);
            foundDesignation = 1;
        }
    }
    if (!foundDesignation) printf("None");
    printf("\n");
}

int main() {
    int n;

    // Input section [cite: 7, 9]
    printf("Enter number of employees: ");
    scanf("%d", &n);

    struct EMPLOYEE emp[n];

    for (int i = 0; i < n; i++) {
        printf("\nEnter details for Employee %d:\n", i + 1);
        printf("Name: ");
        scanf("%s", emp[i].name);
        printf("Designation: ");
        scanf("%s", emp[i].designation);
        printf("Gender: ");
        scanf("%s", emp[i].gender);
        printf("Date of Joining (dd-mm-yyyy): ");
        scanf("%s", emp[i].doj);
        printf("Salary: ");
        scanf("%f", &emp[i].salary);
    }

    // Output section [cite: 8, 9]
    displayStats(emp, n);

    return 0;
}
