#include <stdio.h>
#include <string.h>

int main() {
    char str1[100], str2[100], temp[200];
    int choice;

    printf("Enter first string: ");
    gets(str1); // Note: fgets is safer, but gets matches simple logic
    printf("Enter second string: ");
    gets(str2);

    do {
        printf("\n--- String Operations Menu ---");
        printf("\n1. Length of Strings");
        printf("\n2. Copy String");
        printf("\n3. Concatenation");
        printf("\n4. Comparison");
        printf("\n5. Reverse (Manual)");
        printf("\n6. Exit");
        printf("\nEnter your choice: ");
        scanf("%d", &choice);
        getchar(); // To consume the newline character left by scanf

        switch (choice) {
            case 1:
                // strlen() finds the length of the string [cite: 7]
                printf("Length of first string: %ld\n", strlen(str1));
                printf("Length of second string: %ld\n", strlen(str2));
                break;

            case 2:
                // strcpy() copies source string to destination [cite: 7]
                strcpy(temp, str1);
                printf("First string copied to temp: %s\n", temp);
                break;

            case 3:
                // strcat() appends the second string to the first [cite: 7, 12]
                strcpy(temp, str1); // Using temp to keep original str1 intact
                strcat(temp, str2);
                printf("Concatenated string: %s\n", temp);
                break;

            case 4:
                // strcmp() compares two strings [cite: 7]
                if (strcmp(str1, str2) == 0)
                    printf("Strings are equal.\n");
                else
                    printf("Strings are not equal.\n");
                break;

            case 5:
                // Using strrev (if available) or manual reverse for portability
                printf("First string reversed: ");
                for (int i = strlen(str1) - 1; i >= 0; i--) {
                    putchar(str1[i]);
                }
                printf("\n");
                break;

            case 6:
                printf("Exiting program...\n");
                break;

            default:
                printf("Invalid choice! Please try again.\n");
        }
    } while (choice != 6);

    return 0;
}
