#include <stdio.h>
#include <conio.h>

//Iterative method to calculate factorial
long long factorialIterative(int n)
{
    long long factorial = 1;
    for(int i = 1; i <= n; i++) {
        factorial *= i;
    }
    return factorial;
}

//Recursive method to calculate factorial
long long factorialRecursive(int n)
{
    if(n == 0 || n == 1) {
        return 1;
    }
    return n * factorialRecursive(n - 1);
}

int main()
{
    int number;
    printf("Enter a positive integer: ");
    scanf("%d", &number);

    if(number < 0) {
        printf("Factorial is not defined for negative numbers.\n");
        return 1;
    }

    printf("Factorial of %d (Iterative) = %lld\n", number, factorialIterative(number));
    printf("Factorial of %d (Recursive) = %lld\n", number, factorialRecursive(number));

    return 0;
}
