#include <stdio.h>

int main() {
    int choice;
    double num1, num2, result;
    long long fact = 1;
    int i, n;

    while (1) {
        // Displaying the menu
        printf("\n--- Simple Calculator Menu ---\n");
        printf("1. Addition (+)\n");
        printf("2. Subtraction (-)\n");
        printf("3. Multiplication (*)\n");
        printf("4. Division (/)\n");
        printf("5. Power (x^y)\n");
        printf("6. Factorial (x!)\n");
        printf("7. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        if (choice == 7) {
            printf("Exiting program...\n");
            break;
        }

        switch (choice) {
            case 1: // Addition [cite: 6]
                printf("Enter two numbers: ");
                scanf("%lf %lf", &num1, &num2);
                result = num1 + num2;
                printf("Result = %.2f\n", result);
                break;

            case 2: // Subtraction [cite: 6]
                printf("Enter two numbers: ");
                scanf("%lf %lf", &num1, &num2);
                result = num1 - num2;
                printf("Result = %.2f\n", result);
                break;

            case 3: // Multiplication [cite: 6]
                printf("Enter two numbers: ");
                scanf("%lf %lf", &num1, &num2);
                result = num1 * num2;
                printf("Result = %.2f\n", result);
                break;

            case 4: // Division [cite: 6]
                printf("Enter two numbers: ");
                scanf("%lf %lf", &num1, &num2);
                if (num2 != 0) {
                    result = num1 / num2;
                    printf("Result = %.2f\n", result);
                } else {
                    printf("Error: Division by zero is not allowed!\n");
                }
                break;

            case 5: // Power (x^y) using a loop 
                printf("Enter base (x) and exponent (y): ");
                scanf("%lf %d", &num1, &n);
                result = 1.0;
                for (i = 1; i <= n; i++) {
                    result *= num1;
                }
                printf("Result = %.2f\n", result);
                break;

            case 6: // Factorial (x!) using a loop [cite: 7, 12]
                printf("Enter an integer: ");
                scanf("%d", &n);
                if (n < 0) {
                    printf("Error: Factorial of negative numbers doesn't exist.\n");
                } else {
                    fact = 1;
                    for (i = 1; i <= n; i++) {
                        fact *= i;
                    }
                    printf("Result = %lld\n", fact);
                }
                break;

            default:
                printf("Invalid choice! Please try again.\n");
        }
    }

    return 0;
}
