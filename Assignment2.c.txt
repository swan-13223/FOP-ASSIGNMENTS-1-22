#include <stdio.h>
#include <conio.h>

int main()
{
    int x, a, b, c=0;
    printf("Enter the integers:");
    scanf("%d%d", &a, &b);
    printf("Enter 1 for addition, Enter 2 for subtraction, Enter 3 for multiplication, Enter 4 for division:");
    scanf("%d", &x);
    switch(x)
    {
        case 1:
        c = a+b;
        break;
        case 2:
        c = a-b;
        break;
        case 3:
        c = a*b;
        break;
        case 4:
        c = a/b;
        break;
    }
    printf("%d", c);
    return 0;
}
