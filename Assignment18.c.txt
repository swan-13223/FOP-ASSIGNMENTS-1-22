#include <stdio.h>
#include <string.h>

int main() {
    char str1[100], str2[100], rev[100];
    int i, j, len = 0, flag = 0;

    // 1. Input Primary String
    printf("Enter a string: ");
    scanf("%s", str1);

    // i) Calculate length of string [cite: 1, 5]
    while (str1[len] != '\0') {
        len++;
    }
    printf("Length of string = %d\n", len);

    // ii) String reversal [cite: 1, 5]
    j = 0;
    for (i = len - 1; i >= 0; i--) {
        rev[j] = str1[i];
        j++;
    }
    rev[j] = '\0'; // Null terminate the reversed string 
    printf("Reversed string = %s\n", rev);

    // iv) Check palindrome [cite: 1, 3]
    flag = 0;
    for (i = 0; i < len; i++) {
        if (str1[i] != rev[i]) {
            flag = 1;
            break;
        }
    }
    if (flag == 0)
        printf("The string is a Palindrome\n");
    else
        printf("The string is not a Palindrome\n");

    // 2. Input Second String for Comparison/Substring [cite: 7]
    printf("Enter another string for equality and substring check: ");
    scanf("%s", str2);

    // iii) Equality check of two strings [cite: 1, 5]
    int len2 = 0;
    while (str2[len2] != '\0') len2++;

    flag = 0;
    if (len != len2) {
        flag = 1;
    } else {
        for (i = 0; i < len; i++) {
            if (str1[i] != str2[i]) {
                flag = 1;
                break;
            }
        }
    }

    if (flag == 0)
        printf("Strings are equal\n");
    else
        printf("Strings are not equal\n");

    // v) Check substring [cite: 1, 12]
    char *ptr = strstr(str1, str2);
    if (ptr != NULL)
        printf("Substring found in main string\n");
    else
        printf("Substring not found\n");

    return 0;
}
