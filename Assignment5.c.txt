#include <stdio.h>
#include <conio.h>

#define MAX 10
[span_1](start_span)//Function to read a matrix from user input[span_1]
(end_span)
void readMatrix(int matrix[MAX][MAX], int rows, int cols) {
    printf("Enter the elements of the matrix(%d x %d):\n", rows, cols);
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }
}

[span_2](start_span)//Function to display the matrix[span_2]
(end_span)
void displayMatrix(int matrix[MAX][MAX], int rows, int cols) {
    printf("The matrix is:\n");
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            printf("%d ", matrix[i][j]);
        }
        printf("\n");
    }
}

[span_3](start_span)//Function to calculate the sum of the matrix elements[span_3]
(end_span)
int sumMatrix(int matrix[MAX][MAX], int rows, int cols) {
    int sum = 0;
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            sum += matrix[i][j];
        }
    }
    return sum;
}

[span_4](start_span)//Function to calculate the saddle point of the matrix[span_4]
(end_span)
void saddlePoint(int matrix[MAX][MAX], int rows, int cols) {
    for(int i = 0; i < rows; i++) {
        int minRow = matrix[i][0];
        int colIndex = 0;
        for(int j = 1; j < cols; j++) {
            if(matrix[i][j] < minRow) {
                minRow = matrix[i][j];
                colIndex = j;
            }
        }
        int maxCol = minRow;
        for(int k = 0; k < rows; k++) {
            if(matrix[k][colIndex] > maxCol) {
                maxCol = matrix[k][colIndex];
            }
        }
        if(minRow == maxCol) {
            printf("Saddle point found at (%d, %d): %d\n", i, colIndex, minRow);
            return;
        }
    }
    printf("No saddle point found in the matrix.\n");
    return
}

[span_5](start_span)//function to calculate inverse of a 3x3 matrix[span_5]
(end_span)
void inverseMatrix(int matrix[MAX][MAX], int inverse[MAX][MAX]) {
    int determinant = matrix[0][0]*(matrix[1][1]*matrix[2][2] - matrix[1][2]*matrix[2][1]) -
                      matrix[0][1]*(matrix[1][0]*matrix[2][2] - matrix[1][2]*matrix[2][0]) +
                      matrix[0][2]*(matrix[1][0]*matrix[2][1] - matrix[1][1]*matrix[2][0]);
    
    if(determinant == 0) {
        printf("Inverse does not exist for the given matrix.\n");
        return;
    }
    
    int adjoint[MAX][MAX];
    adjoint[0][0] =  (matrix[1][1]*matrix[2][2] - matrix[1][2]*matrix[2][1]);
    adjoint[0][1] = -(matrix[1][0]*matrix[2][2] - matrix[1][2]*matrix[2][0]);
    adjoint[0][2] =  (matrix[1][0]*matrix[2][1] - matrix[1][1]*matrix[2][0]);
    adjoint[1][0] = -(matrix[0][1]*matrix[2][2] - matrix[0][2]*matrix[2][1]);
    adjoint[1][1] =  (matrix[0][0]*matrix[2][2] - matrix[0][2]*matrix[2][0]);
    adjoint[1][2] = -(matrix[0][0]*matrix[2][1] - matrix[0][1]*matrix[2][0]);
    adjoint[2][0] =  (matrix[0][1]*matrix[1][2] - matrix[0][2]*matrix[1][1]);
    adjoint[2][1] = -(matrix[0][0]*matrix[1][2] - matrix[0][2]*matrix[1][0]);
    adjoint[2][2] =  (matrix[0][0]*matrix[1][1] - matrix[0][1]*matrix[1][0]);

    for(int i = 0; i < 3; i++) {
        for(int j = 0; j < 3; j++) {
            inverse[i][j] = adjoint[i][j] / determinant;
        }
    }
}

[span_6](start_span)//Function to calculate the magic square of matrix[span_6]
(end_span)

void magicSquare(int n) {
    if(n % 2 == 0) {
        printf("Magic square is only possible for odd order matrices.\n");
        return;
    }
    
    int magic[MAX][MAX] = {0};
    int num = 1, i = 0, j = n / 2;

    while(num <= n*n) {
        magic[i][j] = num++;
        int newi = (i - 1 + n) % n;
        int newj = (j + 1) % n;

        if(magic[newi][newj]) {
            i++;
        } else {
            i = newi;
            j = newj;
        }
    }

    printf("The %d x %d Magic Square is:\n", n, n);
    for(i = 0; i < n; i++) {
        for(j = 0; j < n; j++) {
            printf("%d ", magic[i][j]);
        }
        printf("\n");
    }
}
